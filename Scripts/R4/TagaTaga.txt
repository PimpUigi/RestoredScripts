sub ObjectPlayerInteraction
	if Object.PropertyValue==0
		switch Object.State
		case 0
			PlayerObjectCollision(C_TOUCH,-128,-32,0,8)
			if CheckResult==1
				Object.State=2
				Object.Value0=0
				Object.Priority=1
			endif
			break
		case 1
			PlayerObjectCollision(C_TOUCH,0,-32,128,8)
			if CheckResult==1
				Object.State=3
				Object.Value0=0
				Object.Priority=1
			endif
			break
		endswitch
	endif
	if Object.State<4
		PlayerObjectCollision(C_TOUCH,-12,-8,12,8)
		if CheckResult==1
			if Object.Direction==FACING_LEFT
				CallFunction(PlayerObject_BadnikBreak)
			else
				CallFunction(PlayerObject_BadnikBreakFlip)
			endif
		endif
	endif
endsub
